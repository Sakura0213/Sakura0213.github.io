  3. 样式

    <!--内联样式-->
<p style="background-color:green;">这是一个段落</p>

<!--内部样式表-->
<!--在<head>区域使用<style>元素来包含CSS-->
<head>
    <style type="text/css">
      body {
        background-color: yellow;
      }
      p {
        color: blue;
      }
    </style>
</head>

<!--外部样式表-->
<head>
	<link rel="stylesheet" type="text/css" href="mystyle.css">
</head>
